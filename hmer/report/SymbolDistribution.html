def parse_label_string(lbl_str):
    lbl = lbl_str.split()
    img = lbl[0].split('/')[-1]
    lbl = np.asarray(lbl[1:]).astype(np.float32).reshape((-1, 5)).tolist()
    lbl = [bbox[1:] + [int(bbox[0])] for bbox in lbl]
    return img, lbl
